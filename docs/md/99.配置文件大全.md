# 配置文件大全

`application.properties`
```properties
# 默认 web 端口
server.port=7070

# 是否启用debug模式，开启后将会打印比较详细的日志
debug=false

# 静态资源路径 位于 resources/static 下
server.static=/static
# thymeleaf 模板引擎配置
server.template.suffix=.html
# 模板引擎目录 位于 resources/templates 下
server.template.prefix=templates
# 是否使用缓存，生产环境建议启用
server.template.cache=false
# 默认1小时，单位毫秒。缓存时间，server.template.cache为true才有效
server.template.cacheTime=3600000


# session相关
server.session.name=sid
# session 有效时间为毫秒
server.session.age=1800000

# 文件相关
# 默认 3MB 大小, netty HttpObjectAggregator聚合内容的最大长度（字节）, -1或者0 表示 Integer.MAX_VALUE 即2GB内容，netty的限制，超出的大小需要考虑分段上传文件
server.maxContentLength=3145728
# 默认3MB，上传文件内存缓冲大小，超出该大小后会将文件存放到临时目录，会话处理完毕后自动删除。
server.uploadFileBuffer=3145728
# 默认1MB，文件断点续传起步大小，若返回的文件大于此大小，则http将能够使用断点续传，太大会占用运行内存
server.file.ftpSize=1048576

# 缓存相关
# 文件处理缓存，对应响应头的 Cache-Control。默认值为:max-age=1800(30分钟)；可选：no-cache(不缓存)、以及其他http标准值
server.file.cacheControl=max-age=1800

# 日志配置文件
log.file=logback.xml

# 线程相关
# 接收线程数，不需要太大，压力全在处理线程中，0为默认：系统的核数
server.thread.maxReceive=0
# 处理线程数，即处理最大并发数，0为默认：系统的核数 * 50（默认值不会超过200），对比，springboot默认为200
server.thread.maxHandler=0
# 积压队列长度：当达到最大并发数，可以积压的请求数量，默认：256
server.thread.backlog=256

# websocket相关
# 处理消息最大值
websocket.maxMessage=65536
# 超时
websocket.timeout=12000
```
