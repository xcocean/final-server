# 07.日志配置.md


日志默认使用 `logback` 

## 1、默认配置文件 

`resouces/logback.xml` 内容如下：

```xml
<?xml version="1.0" encoding="UTF-8"?>
<configuration>
    <appender name="console" class="ch.qos.logback.core.ConsoleAppender">
        <encoder>
            <pattern>%d{yyyy-MM-dd HH:mm:ss.SSS} %highlight(%5level) [%thread] %cyan(%logger{39} line:[%line]) %msg%n</pattern>
        </encoder>
    </appender>

    <!-- 设置默认日志级别为INFO -->
    <root level="info">
        <appender-ref ref="console"/>
    </root>
</configuration>
```

即加载配置文件的 `log.file=logback.xml` 配置

## 2、指定日志配置文件

`application.properties`中配置 `log.file=logback.xml` 即可

也可以直接配置绝对路径:

`log.file=C:/Users/Administrator/Desktop/logback.xml`

配置文件优先级：`绝对路径 > 相对路径 > resource下`

代码指定(优先级最高)：
```java
FinalServerLogConfig.init("C:\\Users\\Administrator\\Desktop\\logback.xml");
```

## 3、自定义logback配置

可以创建 `resouces/logback.xml` 来覆盖原配置。

### 3.1、参考配置一: 用户目录下

将日志打印到用户目录的logs文件夹下：`resouces/logback.xml`
```xml
<?xml version="1.0" encoding="UTF-8"?>
<configuration>
    <appender name="console" class="ch.qos.logback.core.ConsoleAppender">
        <encoder>
            <pattern>%d{yyyy-MM-dd HH:mm:ss.SSS} [%thread] %-5level %logger{36} - %msg%n</pattern>
        </encoder>
    </appender>

    <!-- info日志 -->
    <appender name="fileInfo" class="ch.qos.logback.core.rolling.RollingFileAppender">
        <file>${HOME:-${USERPROFILE}}/logs/info.log</file>
        <!-- 过滤日志 -->
        <filter class="ch.qos.logback.classic.filter.LevelFilter">
            <level>INFO</level>
            <onMatch>ACCEPT</onMatch>
            <onMismatch>DENY</onMismatch>
        </filter>
        <encoder>
            <pattern>%d{yyyy-MM-dd HH:mm:ss.SSS} [%thread] %-5level %logger{36} - %msg%n</pattern>
        </encoder>
        <!-- 滚动策略 -->
        <rollingPolicy class="ch.qos.logback.core.rolling.SizeAndTimeBasedRollingPolicy">
            <!-- 每天滚动-->
            <fileNamePattern>${HOME:-${USERPROFILE}}/logs/info-%d{yyyy-MM-dd}.%i.log.zip</fileNamePattern>
            <!-- 每个文件最多 10MB，保留 60 天的历史记录，但最多 512MB -->
            <maxFileSize>10MB</maxFileSize>
            <maxHistory>60</maxHistory>
            <totalSizeCap>512MB</totalSizeCap>
        </rollingPolicy>
    </appender>

    <!-- warn日志 -->
    <appender name="fileWarn" class="ch.qos.logback.core.rolling.RollingFileAppender">
        <file>${HOME:-${USERPROFILE}}/logs/warn.log</file>
        <!-- 过滤日志 -->
        <filter class="ch.qos.logback.classic.filter.LevelFilter">
            <level>WARN</level>
            <onMatch>ACCEPT</onMatch>
            <onMismatch>DENY</onMismatch>
        </filter>
        <encoder>
            <pattern>%d{yyyy-MM-dd HH:mm:ss.SSS} [%thread] %-5level %logger{36} - %msg%n</pattern>
        </encoder>
        <!-- 滚动策略 -->
        <rollingPolicy class="ch.qos.logback.core.rolling.SizeAndTimeBasedRollingPolicy">
            <!-- 每天滚动-->
            <fileNamePattern>${HOME:-${USERPROFILE}}/logs/warn-%d{yyyy-MM-dd}.%i.log.zip</fileNamePattern>
            <!-- 每个文件最多 10MB，保留 60 天的历史记录，但最多 512MB -->
            <maxFileSize>10MB</maxFileSize>
            <maxHistory>60</maxHistory>
            <totalSizeCap>512MB</totalSizeCap>
        </rollingPolicy>
    </appender>

    <!-- error日志 -->
    <appender name="fileError" class="ch.qos.logback.core.rolling.RollingFileAppender">
        <file>${HOME:-${USERPROFILE}}/logs/error.log</file>
        <!-- 过滤日志 -->
        <filter class="ch.qos.logback.classic.filter.LevelFilter">
            <level>ERROR</level>
            <onMatch>ACCEPT</onMatch>
            <onMismatch>DENY</onMismatch>
        </filter>
        <encoder>
            <pattern>%d{yyyy-MM-dd HH:mm:ss.SSS} [%thread] %-5level %logger{36} - %msg%n</pattern>
        </encoder>
        <!-- 滚动策略 -->
        <rollingPolicy class="ch.qos.logback.core.rolling.SizeAndTimeBasedRollingPolicy">
            <!-- 每天滚动-->
            <fileNamePattern>${HOME:-${USERPROFILE}}/logs/error-%d{yyyy-MM-dd}.%i.log.zip</fileNamePattern>
            <!-- 每个文件最多 10MB，保留 60 天的历史记录，但最多 512MB -->
            <maxFileSize>10MB</maxFileSize>
            <maxHistory>60</maxHistory>
            <totalSizeCap>512MB</totalSizeCap>
        </rollingPolicy>
    </appender>

    <root level="DEBUG">
        <appender-ref ref="console"/>
    </root>

    <!-- 设置默认日志级别为INFO -->
    <root level="info">
        <appender-ref ref="fileInfo"/>
        <appender-ref ref="fileWarn"/>
        <appender-ref ref="fileError"/>
    </root>
</configuration>
```
