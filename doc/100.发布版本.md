## 2023-01-08
`1.0.9-SNAPSHOT`
发布快照版。若要使用快照版时，需要添加远程快照仓库：
```xml
<repositories>
    <repository>
        <id>nexus</id>
        <name>Nexus</name>
        <layout>default</layout>
        <url>https://s01.oss.sonatype.org/content/repositories/snapshots</url>
        <snapshots>
            <enabled>true</enabled>
        </snapshots>
        <releases>
            <enabled>true</enabled>
        </releases>
    </repository>
</repositories>
```
